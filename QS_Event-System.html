Event Manager:

The game event manager takes all event listeners and adds their events to the dictionary. When an event is invoked the event manager will check if it's in the dictionary. If it's in the dictionary the event manager will then invoke it.

Code:

public void ListenFor(string eventKey, UnityEngine.Object bindingObject, Action action)
{
	// Checks whether dictionary already has the eventKey
	if (!EventDictionary.ContainsKey(eventKey))
		EventDictionary.Add(eventKey, new());

	// Adds the event to the dictionary
	EventDictionary[eventKey].AddListener(() => SafeInvoke(bindingObject, action));
}
public void InvokeEvent(string eventKey)
{
	if (EventDictionary.ContainsKey(eventKey))
		EventDictionary[eventKey].Invoke();
}

Event Listener:

Code:

private void Start()
{
	GameEventManager.Instance.ListenFor(_eventName, this, _event.Invoke);
}

Event Invoker:

Code:

private void OnTriggerEnter(Collider other)
{
	if (!_isInvokable)
		return;

	if (!_collidables.LayerInMask(other.gameObject.layer))
		return;

	GameEventManager.Instance.InvokeEvent(_eventKey);

	_timesInvoked++;

	if (_timesInvoked >= _allowedInvokes)
		_isInvokable = false;
}
