private IEnumerator TypeLine()
{
	SetFontStyles();

	int characterCount = 0;
	string input = _currentDialogueData[_index].Text;

	// Constructs a string to highlight a part of the text
	string output = input.Replace("[","<color=#" + ColorUtility.ToHtmlStringRGB(_currentDialogueData[_index].HighlightColor) + ">");
	output = output.Replace("]", "<color=#" + ColorUtility.ToHtmlStringRGB(_baseTextColor) + ">");

	_textComponent.text = output;

	// Types out the string 
	foreach (char c in output.ToCharArray())
	{
		// Adds to the maxVisibleCharacters make it seem like it's typing the text out
		characterCount++;
		_textComponent.maxVisibleCharacters = characterCount;
		yield return new WaitForSeconds(_textSpeed);
	}
	yield return new WaitForSeconds(_pauseBetweenLines);

	NextLine();
}
